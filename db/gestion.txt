-- Crear la base de datos
CREATE DATABASE IF NOT EXISTS gestion;
USE gestion;

-- Crear la tabla administradores
CREATE TABLE administradores (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    nombres VARCHAR(255) NOT NULL,
    apellidos VARCHAR(255) NOT NULL,
    correo VARCHAR(255) NOT NULL,
    contrasena VARCHAR(255) NOT NULL,
    dni VARCHAR(20) NOT NULL,
    celular VARCHAR(20) NOT NULL,
    fecha_nacimiento DATE NOT NULL,
    direccion TEXT NOT NULL,
    ocupacion VARCHAR(255) NOT NULL
);

-- Crear la tabla categorias
CREATE TABLE categorias (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL
);

-- Crear la tabla consumibles
CREATE TABLE consumibles (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    descripcion_consumible TEXT,
    unidad_medida VARCHAR(255) NOT NULL,
    categoria_id INT(11),W
    precio_sugerido DECIMAL(10,2),
    es_compuesto TINYINT(1) DEFAULT 0,
    marca VARCHAR(255),
    observacion TEXT,
    CONSTRAINT fk_categoria FOREIGN KEY (categoria_id) REFERENCES categorias(id)
);

-- Crear la tabla consumibles_categorias
CREATE TABLE consumibles_categorias (
    consumible_id INT(11) NOT NULL,
    categoria_id INT(11) NOT NULL,
    PRIMARY KEY (consumible_id, categoria_id),
    CONSTRAINT fk_consumible_categoria FOREIGN KEY (consumible_id) REFERENCES consumibles(id),
    CONSTRAINT fk_categoria_consumible FOREIGN KEY (categoria_id) REFERENCES categorias(id)
);

-- Crear la tabla compras_consumibles
CREATE TABLE compras_consumibles (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    consumible_id INT(11) NOT NULL,
    cantidad INT(11) NOT NULL,
    costo_unitario DECIMAL(10,2) NOT NULL,
    total DECIMAL(10,2) NOT NULL,
    fecha_ingreso DATE NOT NULL,
    fecha_vencimiento DATE NOT NULL,
    metodo_pago VARCHAR(40),
    CONSTRAINT fk_consumible FOREIGN KEY (consumible_id) REFERENCES consumibles(id)
);

-- Crear la tabla lotes
CREATE TABLE lotes (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    compras_consumibles_id INT(11) NOT NULL,
    lote VARCHAR(255) NOT NULL,
    cantidad INT(11) NOT NULL,
    costo_total DECIMAL(10,2) NOT NULL,
    precio_unitario DECIMAL(10,2) NOT NULL,
    fecha_ingreso DATE NOT NULL,
    fecha_vencimiento DATE NOT NULL,
    CONSTRAINT fk_compras_consumibles FOREIGN KEY (compras_consumibles_id) REFERENCES compras_consumibles(id)
);

-- Crear la tabla consumible_componentes
CREATE TABLE consumible_componentes (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    id_consumible INT(11) NOT NULL,
    id_componente INT(11) NOT NULL,
    cantidad DECIMAL(10,2) NOT NULL,
    CONSTRAINT fk_consumible_compuesto FOREIGN KEY (id_consumible) REFERENCES consumibles(id),
    CONSTRAINT fk_componente_simple FOREIGN KEY (id_componente) REFERENCES consumibles(id)
);

-- Crear la tabla compras_normales
CREATE TABLE compras_normales (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    descripcion_compra VARCHAR(255) NOT NULL,
    cantidad INT(11) NOT NULL,
    costo_unitario DECIMAL(10,2) NOT NULL,
    total DECIMAL(10,2) NOT NULL,
    fecha DATE NOT NULL DEFAULT CURDATE(),
    proveedor VARCHAR(255),
    metodo_pago VARCHAR(255) NOT NULL,
    observacion TEXT
);

-- Crear la tabla ventas
CREATE TABLE ventas (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    total DECIMAL(10,2) NOT NULL,
    metodo_pago ENUM('Efectivo','Visa','Yape','Plin') NOT NULL,
    fecha DATE NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP()
);

-- Crear la tabla ventas_detalles
CREATE TABLE ventas_detalles (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    venta_id INT(11) NOT NULL,
    consumible_id INT(11) NOT NULL,
    cantidad INT(11) NOT NULL,
    precio_unitario DECIMAL(10,2) NOT NULL,
    CONSTRAINT fk_venta FOREIGN KEY (venta_id) REFERENCES ventas(id),
    CONSTRAINT fk_consumible_venta FOREIGN KEY (consumible_id) REFERENCES consumibles(id)
);

-- Tabla de clientes (como referencia)
CREATE TABLE clientes (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    descipcion TEXT
);

-- Tabla de renta (como referencia)
CREATE TABLE renta (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    descipcion TEXT
);
